git config --global -e --> Abrir editor

------------------------------------------------------------------------------------------
ls --> Listar todos los archivos dentro de un directorio
ls -a --> Listar directorios y archivos ocultos
cd 'NombreDirectorio' --> Moverse entre directorios
cd .. --> Volver al directorio anterior
mkdir 'NombreDirectorio' --> Crear directorio
git init --> Iniciar repositorio en git
git status --> Mostrar el estado actual del repositorio
git add 'NombreArchivo'--> Añadir actualizacion al repositorio, cuando se realiza un cambio hay que añadirlo de nuevo.
git commit -m "'Mensaje que tenga sentido'" --> Comprometer cambios, es decir guardarlos tras aplicarse el git add.
rm 'NombreArchivo' --> Eliminar el archivo del repositorio
restore 'NombreArchivo' --> Restaurar archivo
mv 'NombreArchivo' 'NuevoNombreArchivo' --> Cambiar el nombre del archivo
